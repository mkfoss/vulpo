f4len

                 Usage:    unsigned f4len( const FIELD4 *field )

          Description:    The length of the field, in bytes, is returned. This is the length specified for the field when the data file was originally created.

/*ex97.c*/

#include "d4all.h"

 

char *createBufCopy( FIELD4 *f )

{

   char *buf ;

 

   buf = (char *) malloc(f4len( f ) +1 ) ;

   memcpy( buf, f4ptr( f ), f4len( f ) );

   buf[f4len( f )] = 0;

   return buf ;

}

 

void main()

{

   CODE4 cb ;

   DATA4 *data;

   FIELD4 *field ;

   char *buffer ;

 

   code4init( &cb ) ;

   data = d4open( &cb, "INFO" ) ;

   field = d4field( data, "NAME" ) ;

   d4top( data ) ;

   buffer = createBufCopy( field ) ;

   printf( "the copy of the buffer is %s\n", buffer ) ;

   code4initUndo( &cb ) ;

}
